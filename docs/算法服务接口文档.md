# 无人机航拍检测系统 - 算法服务接口文档

## 1. 概述

本文档面向算法部门，描述如何实现基于 PyTorch + YOLOv3 的目标检测服务，以及与后端服务的接口规范。

### 1.1 服务架构

```
┌─────────────────┐     HTTP/gRPC      ┌──────────────────┐
│   后端服务       │ ◄────────────────► │   算法服务        │
│  (Spring Boot)  │                    │  (PyTorch/YOLO)  │
└─────────────────┘                    └──────────────────┘
                                              │
                                              ▼
                                        ┌─────────────┐
                                        │  GPU/CPU    │
                                        │  推理引擎    │
                                        └─────────────┘
```

### 1.2 技术栈

- **框架**: PyTorch 2.0+
- **检测模型**: YOLOv3 (支持 YOLOv3-tiny, YOLOv3, YOLOv3-spp)
- **服务框架**: FastAPI / Flask
- **通信协议**: HTTP RESTful API + WebSocket (实时推送)
- **视频处理**: OpenCV, FFmpeg
- **目标追踪**: DeepSORT / ByteTrack (可选)

## 5. 核心实现要求

### 5.1 YOLOv3 检测实现

```python
# 伪代码示例
import torch
import cv2
import numpy as np
from models import Darknet
from utils.utils import non_max_suppression, scale_coords

class YOLOv3Detector:
    def __init__(self, model_path, config_path, device='cuda'):
        self.device = torch.device(device if torch.cuda.is_available() else 'cpu')
        self.model = Darknet(config_path).to(self.device)
        self.model.load_state_dict(torch.load(model_path, map_location=self.device))
        self.model.eval()
        
    def detect(self, image, conf_threshold=0.5, nms_threshold=0.4):
        """
        单帧图像检测
        
        Args:
            image: numpy array (H, W, C) BGR格式
            conf_threshold: 置信度阈值
            nms_threshold: NMS阈值
            
        Returns:
            detections: list of dict
        """
        # 预处理
        img = self.preprocess(image)
        img = torch.from_numpy(img).to(self.device)
        img = img.float() / 255.0
        
        if img.ndimension() == 3:
            img = img.unsqueeze(0)
        
        # 推理
        with torch.no_grad():
            pred = self.model(img)[0]
            pred = non_max_suppression(pred, conf_threshold, nms_threshold)
        
        # 后处理
        detections = []
        if pred[0] is not None:
            det = pred[0]
            det[:, :4] = scale_coords(img.shape[2:], det[:, :4], image.shape).round()
            
            for *xyxy, conf, cls in det:
                x1, y1, x2, y2 = map(int, xyxy)
                detections.append({
                    'class': self.names[int(cls)],
                    'class_id': int(cls),
                    'confidence': float(conf),
                    'bbox': {
                        'x': x1,
                        'y': y1,
                        'width': x2 - x1,
                        'height': y2 - y1
                    }
                })
        
        return detections
    
    def detect_video(self, video_path, frame_interval=1, callback=None):
        """
        视频检测
        
        Args:
            video_path: 视频文件路径
            frame_interval: 检测帧间隔
            callback: 进度回调函数
        """
        cap = cv2.VideoCapture(video_path)
        total_frames = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))
        fps = cap.get(cv2.CAP_PROP_FPS)
        
        frame_count = 0
        results = []
        
        while True:
            ret, frame = cap.read()
            if not ret:
                break
            
            frame_count += 1
            
            # 按间隔检测
            if frame_count % frame_interval == 0:
                detections = self.detect(frame)
                
                result = {
                    'frame_number': frame_count,
                    'timestamp': frame_count / fps,
                    'detection_count': len(detections),
                    'detections': detections
                }
                results.append(result)
                
                # 回调通知
                if callback:
                    callback({
                        'progress': frame_count / total_frames * 100,
                        'current_frame': frame_count,
                        'detection_count': len(detections)
                    })
        
        cap.release()
        return results
```

### 5.3 目标追踪（可选）

如果需要实现目标追踪功能，建议集成 DeepSORT 或 ByteTrack:

```python
from deep_sort import DeepSort

class VideoDetectorWithTracking:
    def __init__(self, detector, tracker):
        self.detector = detector
        self.tracker = tracker  # DeepSORT 或 ByteTrack 实例
        
    def detect_and_track(self, frame):
        # 检测
        detections = self.detector.detect(frame)
        
        # 转换为追踪器输入格式
        bbox_xywh = []
        confidences = []
        for det in detections:
            x, y, w, h = det['bbox'].values()
            bbox_xywh.append([x + w/2, y + h/2, w, h])  # 中心点坐标
            confidences.append(det['confidence'])
        
        # 追踪
        if len(bbox_xywh) > 0:
            bbox_xywh = torch.Tensor(bbox_xywh)
            confidences = torch.Tensor(confidences)
            outputs = self.tracker.update(bbox_xywh, confidences, frame)
            
            # 关联检测和追踪结果
            for i, output in enumerate(outputs):
                track_id = int(output[4])
                detections[i]['track_id'] = track_id
        
        return detections
```

## 6. 性能优化建议

### 6.1 模型优化

1. **模型量化**: 使用 INT8 量化减少模型大小和推理时间
2. **TensorRT**: 使用 TensorRT 加速推理（推荐）
3. **ONNX**: 转换为 ONNX 格式，使用 ONNX Runtime
4. **半精度**: 使用 FP16 半精度推理

```python
# TensorRT 示例
import torch
from torch2trt import torch2trt

# 创建示例输入
x = torch.ones((1, 3, 608, 608)).cuda()

# 转换模型
model_trt = torch2trt(model, [x])

# 保存
torch.save(model_trt.state_dict(), "yolov3_trt.pth")
```

### 6.2 视频处理优化

1. **多线程解码**: 使用独立线程解码视频
2. **预读取缓冲**: 预读取帧到缓冲区
3. **批量推理**: 多帧一起送入模型推理
4. **异步IO**: 异步保存结果和帧图像

```python
# 批量推理示例
def batch_detect(self, frames, batch_size=4):
    """批量检测多帧"""
    results = []
    
    for i in range(0, len(frames), batch_size):
        batch = frames[i:i+batch_size]
        
        # 批量预处理
        imgs = [self.preprocess(f) for f in batch]
        imgs = np.stack(imgs, axis=0)
        imgs = torch.from_numpy(imgs).to(self.device)
        
        # 批量推理
        with torch.no_grad():
            preds = self.model(imgs)
        
        # 批量后处理
        for j, pred in enumerate(preds):
            detections = self.postprocess(pred, batch[j].shape)
            results.append(detections)
    
    return results
```

| TASK_NOT_FOUND | 任务不存在 | 检查任务ID是否正确 |
